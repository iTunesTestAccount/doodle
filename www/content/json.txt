@Doodle::JSON@ adds the {{{#to_json}}} method:

<<< xmp
#:requires
require 'rubygems'
require 'doodle'
require 'doodle/json'
#:definitions
class Name < Doodle
  has :value, :kind => String
end
class Age < Doodle
  has :value, :kind => Integer
end
class Person < Doodle
  has Name
  has Age
end
#:use
person = Person do
  name "Corum"
  age 999
end

puts person.to_json
#:output
>>>
